IObserver

命名空间: Global

观察者接口，用于实现观察者模式。配合Notice类使用，接收通知消息。

← 返回首页

接口方法

void OnNotice(string notice, params object[] param)
接收通知时调用

使用示例

public class MyObserver : IObserver
{
    public void OnNotice(string notice, params object[] param)
    {
        if (notice == "OnCreatureEscape")
        {
            CreatureModel creature = param[0] as CreatureModel;
            Debug.Log("异想体逃脱: " + creature.name);
        }
    }
}

// 订阅通知
MyObserver observer = new MyObserver();
Notice.instance.Observe("OnCreatureEscape", observer);

// 取消订阅
Notice.instance.Remove("OnCreatureEscape", observer);
